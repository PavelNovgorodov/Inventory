# Алгоритм работы командлета
 
 ```mermaid
flowchart TD
    Start(["Старт <br>Start-Inventory.psm1"])
    LoadModules[Загрузка скриптов из каталога Modules]
    LogFile[["Проверка возможности ведения Log файла
              CheckLogFile.ps1"]]
    InvFolderAnyComputers[["Проверка каталога размещения инвентаризационной информации
                            CheckFolder.ps1"]]
    ModuleParameters[[Обработка дополнительных параметров модулей]]
    StartLocal[["Запуск локального сбора
                 InvLocal.ps1"]]
    StartRemote[[Запуск удаленного сбора]]
    StartResult[[Запуск процедуры обработки]]
    Result-Inventory[/Сохранение инвентаризационной информации/]
    Result-Create[/Сохранение результатов обработки/]
    Finish([Финиш])

    Start --> LoadModules
    LoadModules --> LogFile
    LogFile --> InvFolderAnyComputers
    InvFolderAnyComputers --> ModuleParameters
    ModuleParameters --> Case{"Тип запуска
                               StartType"}
    Case --> |InvLocal| StartLocal
    Case --> |InvRemote| StartRemote
    Case --> |CreateResult| StartResult
    StartLocal -->  Result-Inventory
    StartRemote -->  Result-Inventory
    StartResult -->  Result-Create
    Result-Inventory --> Finish
    Result-Create --> Finish
```
